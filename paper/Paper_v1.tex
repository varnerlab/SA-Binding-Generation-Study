\noindent\fbox{\parbox{\textwidth}{\small\textbf{Significance Statement.}
Protein engineers routinely isolate a handful of functional binders---via phage display, yeast
surface display, or functional selections---but lack a systematic way to expand these small sets
into diverse candidate libraries without retraining a generative model. We show that a single
scalar parameter added to the softmax logits of a Hopfield-energy sampler continuously conditions
generation from ``any family member'' to ``only binders,'' with no training, no GPU, and no
architectural changes. The effectiveness of this conditioning is predicted \emph{a priori} by a
cheap geometric property of the alignment, giving practitioners a decision rule before any
sequences are generated.}}

  \caption{\textbf{Phenotype fidelity is invariant to input set size; diversity
  and composition fidelity improve with more binders.}
  Subsampling $K_{\mathrm{des}} \in \{3,5,8,10,15,20,25,32\}$ sequences from
  the Kunitz strong-binder set (3 replicates each).
  \textit{Left:} P1 K/R fraction remains at 1.0 at every sample size,

  \textit{Centre:} Pairwise sequence diversity increases from 0.36

  \textit{Right:} KL divergence of amino acid composition decreases from 0.047

  phenotype transfer; more binders improve the compositional and diversity
  quality of the generated library.}

% -----------------------------------------------------------------------
% Fig 4: Phase transition entropy curves
% -----------------------------------------------------------------------
\begin{figure}[p]
  \centering
  \includegraphics[width=0.82\textwidth]{sections/figs/fig5_entropy_curves.pdf}
  \caption{\textbf{Phase transition shifts rightward with increasing multiplicity
  ratio $\rho$.}
  Attention entropy $H(\beta)$ as a function of inverse temperature $\beta$
  (log scale) for $\rho \in \{1, 5, 20, 100, 1000\}$ on the Kunitz family.
  Each curve shows a sigmoidal drop from high entropy (exploration phase) to
  near-zero entropy (retrieval phase); dashed vertical lines mark the inflection
  point $\beta^{*}(\rho)$ for each curve.
  As $\rho$ increases, $K_{\mathrm{eff}}$ decreases and $\beta^{*}$ shifts
  rightward (from 4.3 at $\rho=1$ to 10.1 at $\rho=1000$), consistent with
  fewer effective competing attractors requiring higher inverse temperature to
  produce an ordered phase.}
  \label{fig:phase-transition}
\end{figure}

% -----------------------------------------------------------------------
% Fig 5: Calibration gap decomposition
% -----------------------------------------------------------------------
\begin{figure}[p]
  \centering
  \includegraphics[width=0.82\textwidth]{sections/figs/fig2_gap_decomposition.pdf}
  \caption{\textbf{Calibration gap decomposition: three independently measured
  signal layers.}
  Each point corresponds to one $\rho$ value in a fine sweep
  ($\rho \in [1,1000]$, 17 values, 5 replicates).
  Orange diamonds: attention weight on designated binder patterns
  ($\bar{a}_{\mathrm{des}}$) — tracks the ideal diagonal ($y=x$) with
  $<0.3\%$ deviation, confirming that the multiplicity-weighted energy is exact.
  Purple squares: soft P1 K/R score ($f_{\mathrm{soft}}$) after PCA
  reconstruction — nearly flat at $\approx 0.27$ regardless of $\rho$,
  identifying the PCA encoding as the dominant bottleneck.
  Blue circles: hard-decoded P1 K/R fraction ($f_{\mathrm{obs}}$) — lies
  above $f_{\mathrm{soft}}$, indicating that argmax decoding partially recovers
  signal lost in the PCA projection (negative argmax gap).}
  \label{fig:gap-decomposition}
\end{figure}

% Fig 6: Separation predicts gap (money figure)

\begin{figure}[p]
  \centering
  \includegraphics[width=0.72\textwidth]{sections/figs/fig2_separation_vs_gap.pdf}
  \caption{\textbf{Fisher separation index quantitatively predicts the
  calibration gap across protein families.}
  Calibration gap $\Delta = f_{\mathrm{eff}} - f_{\mathrm{obs}}$ at $\rho=500$
  plotted against Fisher separation index $S$ for three families: WW domains
  ($S=0.11$, $\Delta=0.63$), Kunitz domains ($S=0.20$, $\Delta=0.37$), and
  SH3 domains ($S=0.34$, $\Delta \approx 0$). Dashed line: linear fit
  $\Delta \approx 0.84 - 2.5\,S$ ($R^{2} \approx 1.0$). The separation index,
  computable from the alignment alone before any sequences are generated,
  serves as an \emph{a priori} predictor of conditioning effectiveness.}
  \label{fig:cross-family}
\end{figure}
